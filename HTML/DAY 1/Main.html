3. Code Format (C Program)


#include <stdio.h>

int factorial(int n) {
    if (n == 0 || n == 1) {
        return 1;   // Base case
    }
    return n * factorial(n - 1);  // Recursive call
}

int main() {
    int num;

    printf("Enter a number: ");
    scanf("%d", &num);

    int result = factorial(num);

    printf("Factorial of %d is %d\n", num, result);

    return 0;
}
